TWO MIN : SELECT DISTINCT SALARY FROM EMPLOYEE ORDER BY SALARY ASC LIMIT 2; FOR MIN GENERIC : SELECT DISTINCT SALARY FROM EMPLOYEE E1 WHERE 2 >= (SELECT COUNT(DISTINCT SALARY) FROM EMPLOYEE E2 WHERE E1.SALARY >= E2.SALARY) ORDER BY E1.SALARY; TWO MAX : SELECT DISTINCT SALARY FROM EMPLOYEE ORDER BY SALARY DESC LIMIT 2; FOR MAX GENERIC : SELECT DISTINCT SALARY FROM EMPLOYEE E1 WHERE 2 >= (SELECT COUNT(DISTINCT SALARY) FROM EMPLOYEE E2 WHERE E1.SALARY <= E2.SALARY) ORDER BY E1.SALARY;
